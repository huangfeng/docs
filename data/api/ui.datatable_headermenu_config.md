headermenu
=============

@short: adds a headermenu to control column visibility
	

@type: boolean,object
@example:

//simple config
webix.ui({
	view:"datatable", 
    headermenu:true
});

//complex config
webix.ui({
	view:"datatable", 
    headermenu:{
    	width:250, 
        autoheight:false, 
        scroll:true
    }
});
